namespace = painted_galaxy_extra_spawn

# delete empires that spawned without a spawn_weight system
# ie too many empires were created
# checks if the starting system is missing the empire_home_system flag
# extra spawns
event = {
	id = painted_galaxy_extra_spawn.1
	is_triggered_only = yes
	hide_window = yes
	trigger = {
		is_maybe_painted_galaxy = yes
	}
	immediate = {
		every_country = {
			limit = {
				is_country_type = default
				capital_scope.solar_system = { NOT = { has_star_flag = empire_home_system } }
			}
			log = "Painted Galaxy: Deleting extra empire spawn [This.GetName]"
			every_controlled_fleet = { delete_fleet = this }
			every_owned_planet = {
				every_owned_pop_group = { kill_all_pop = yes }
				remove_all_buildings = yes
				destroy_colony = yes
			}
			destroy_country = yes
		}
	}
}
